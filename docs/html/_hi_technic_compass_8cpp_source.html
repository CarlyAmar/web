69 float HiTechnicCompass::GetAngle()
70 {
71  uint16_t heading = 0;

74  m_i2c->Read(kHeadingRegister, sizeof(heading), (uint8_t *)&heading);
75 
76  // Sensor is little endian... swap bytes

78  }
